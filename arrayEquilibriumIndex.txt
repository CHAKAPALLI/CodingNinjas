import java.util.Scanner;

public class Solution {

	public static int arrayEquilibriumIndex(int[] arr) {
		// Your code goes here
		int rightSum = 0, leftSum = 0;

		for (int i = 0; i < arr.length; i++) {
			rightSum += arr[i];
		}

		for (int i = 0; i < arr.length; i++) {
			rightSum -= arr[i];

			if (leftSum == rightSum) {
				return i;
			}
			leftSum += arr[i];
		}
		return -1;
	}
}
