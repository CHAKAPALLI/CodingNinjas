
public class Solution {

    public static void rotate(int[] arr, int d) {
      // Check if rotation is needed
    if (d > 0) {
        // Compute the effective rotation amount
        d = d % arr.length;

        // Create a temporary array to store the first d elements
        int[] temp = new int[d];

        // Copy the first d elements into the temporary array
        for (int i = 0; i < d; i++) {
            temp[i] = arr[i];
        }

        // Shift the remaining elements to the left by d positions
        for (int i = 0, j = 0; i < arr.length && j < d; i++) {
            if (i >= (arr.length - d)) {
                // Copy the elements from the temporary array
                arr[i] = temp[j];
                j++;
            } else {
                // Shift the elements by d positions
                arr[i] = arr[(i + d) % arr.length];
            }
        }
    }
    }

}
